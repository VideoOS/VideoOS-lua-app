scrW, scrH = System:screenSize();

function createButton( text , x,y,w,h, callback) 
	local button = Button();
	button:frame(x,y,w,h);
	button:text(text);
    button:textColor(0x000000)
    button:fontSize(24)
	button:backgroundColor(0xDC143C);
	button:callback( function()
		print("ok",button);
		if( callback ) then
			callback();
		end
	end );
	return button;
end

buttonH = scrH /10;
